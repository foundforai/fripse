  <script>
    const FORMSPREE_ENDPOINT = "https://formspree.io/f/movklzvl";

    const form = document.getElementById("readinessForm");
    const status = document.getElementById("status");
    const submitBtn = document.getElementById("submitBtn");
    const formError = document.getElementById("formError");

    const requiredIds = ["name","email","company","consent"];

    function showFormError(msg) {
      formError.textContent = msg;
      formError.style.display = "block";
    }
    function clearFormError() {
      formError.textContent = "";
      formError.style.display = "none";
    }
    function setInvalid(id, invalid) {
      const el = document.getElementById(id);
      if (!el) return;
      if (invalid) el.classList.add("invalid");
      else el.classList.remove("invalid");
    }

    function validate() {
      clearFormError();
      let ok = true;
      for (const id of requiredIds) {
        const el = document.getElementById(id);
        const valid = id === "consent" ? el.checked : !!el.value.trim();
        setInvalid(id, !valid);
        if (!valid) ok = false;
      }
      if (!ok) showFormError("Please complete all required fields.");
      const email = document.getElementById("email").value.trim();
      if (ok && !/^[^\\s@]+@[^\\s@]+\\.[^\\s@]+$/.test(email)) {
        setInvalid("email", true);
        showFormError("Please enter a valid email address.");
        ok = false;
      }
      return ok;
    }

    form.addEventListener("submit", async (e) => {
      e.preventDefault();

      // Honeypot
      if (document.getElementById("hp_field").value.trim() !== "") {
        sessionStorage.setItem("fripse_quiz_submitted","true");
        window.location.href = "/thank-you.html";
        return;
      }

      if (!validate()) return;

      submitBtn.disabled = true;
      status.textContent = "Submitting, please wait…";

      const formData = new FormData(form);
      formData.append("_subject","New Fripse AI Readiness Report");
      formData.append("_gotcha",""); // extra spam trap commonly supported

      try {
        const res = await fetch(FORMSPREE_ENDPOINT, {
          method: "POST",
          headers: { "Accept": "application/json" },
          body: formData
        });

        if (res.ok) {
          sessionStorage.setItem("fripse_quiz_submitted","true");
          form.reset();
          window.location.href = "/thank-you.html";
        } else {
          const data = await res.json().catch(()=>({}));
          const msg = data.errors ? data.errors.map(e => e.message).join(", ") : "Submission failed, please try again.";
          showFormError(msg);
        }
      } catch (err) {
        showFormError("Network error, please try again.");
      } finally {
        submitBtn.disabled = false;
        status.textContent = "";
      }
    });
  </script>
